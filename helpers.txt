SELECT ressource_kind.name, ressources.name, sum(invoices_items.amount) FROM `invoices_items` join ressources on ressources.id = invoices_items.ressource_id
join invoices on invoices.id = invoices_items.invoice_id
join ressource_kind on ressource_kind.id = ressources.ressource_kind_id
WHERE invoices.organisation_id = 617 and invoices.date_invoice >= "2018-01-01"
and invoices_items.amount < 0
group by ressources.id
